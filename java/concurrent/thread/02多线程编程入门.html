# 二、多线程基础入门

# 1.多线程的创建

# 1.继承Thread类

public class ThreadTest01 {

    public static void main(String[] args) {

        System.out.println("我是主线程，线程id==>"+Thread.currentThread().getName());

        new ThreadDemo().start();

        System.out.println("我是主线程，线程id==>"+Thread.currentThread().getName());

    }

    //新建一个类，继承Thread，重写run方法，即可创建一个新的线程
    static class ThreadDemo extends Thread{
        @Override
        public void run() {
            System.out.println("我是新的线程，线程id==>"+Thread.currentThread().getName());
        }
    }
}

# 2.实现Runnable接口

public static void main(String[] args) {

        System.out.println("我是主线程，线程id==>"+Thread.currentThread().getName());

        //实现Runnable接口方式
        new Thread(new Runnable() {
            @Override
            public void run() {
                System.out.println("我是子线程，线程id==>"+Thread.currentThread().getName());
            }
        }).start();

        System.out.println("我是主线程，线程id==>"+Thread.currentThread().getName());

    }

# 3，Callable方式

//匿名内部类方式创建Callable接口
        Callable<String> callable = new Callable<String>() {
            @Override
            public String call() throws Exception {
                return "Hello";
            }
        };
        //使用FutureTask模式创建线程
        FutureTask<String> future = new FutureTask<String>(callable);

        //新建线程，并启动
        new Thread(future).start();

        //阻塞获取线程返回值
        String s = future.get();
        System.out.println(s);
